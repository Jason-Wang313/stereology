\item[Theorem 2 (Indistinguishability Bound)]~\\
  \textbf{Base:} \texttt{theorem2\_proof.tex}\\

\item[Corollary 2.1 (Ranking Unreliability)]~\\
  \textbf{Source:} \texttt{gap2\_ranking\_unreliability.tex}\\
  Chi-squared selection model with computable formula:\\
  $P(\text{top-1 wrong}) \leq \sum_j \Phi(-\Delta_j / (2\sigma_{\mathrm{hidden}}))$\\
  where $\sigma_{\mathrm{hidden}} = \sqrt{2(D - d_{\mathrm{eff}})}$.\\
  Includes signal-to-noise characterization $\rho = \sqrt{d_{\mathrm{eff}}/D}$.\\
  \textbf{Convexity: not required} (operates on score distributions).

\item[Corollary 2.2 (Rank Reversal)]~\\
  \textbf{Source:} \texttt{theorem2\_proof.tex} (Corollary 2.2)\\
  Pure linear algebra; connects to Belton \& Gear (1983).\\
  \textbf{Convexity: not required.}

\item[Theorem 4 (Tight Stability Bound)]~\\
  \textbf{Planar case:} \texttt{theorem4\_proof.tex}\\
  \textbf{General dimension:} \texttt{gap1\_moonshot\_theorem4\_general.tex}\\
  Minimax optimal rate $m^{-2/(D-1)}$ via optimal recovery theory:
  \begin{itemize}
    \item Upper bound via Kolmogorov $n$-widths and Jackson's inequality on $S^{D-1}$.
    \item Lower bound via spherical polynomial null-space construction in $\mathcal{P}_{2L}$.
    \item Square-root law corollary.
  \end{itemize}
  The planar file is retained for its concrete Fourier argument.\\
  \textbf{Convexity: required} (plus curvature lower bound $\kappa > 0$).

\item[Resolution of Gardner's Problem 1.5]~\\
  \textbf{Source:} \texttt{problem15\_complete.tex}, \texttt{residual1\_resolved.tex}, \texttt{final\_1percent.tex}\\
  Contains:
  \begin{itemize}
    \item \textbf{Curvature--regularity correspondence}: $r_{\max} \sim \mathrm{dist}^{-\beta} \Rightarrow h_K \in C^{1,1-\beta}$.
    \item \textbf{Universal stability theorem}: $\delta_H = \Theta(M \cdot m^{-(1+\alpha)/(D-1)})$ for $C^{1,\alpha}$ support functions, minimax optimal.
    \item \textbf{The $\beta$-rate}: $\delta_H = \Theta(m^{-(2-\beta)/(D-1)})$ for $0 \leq \beta \leq 1$.
    \item \textbf{X-rays equal widths} for non-adaptive measurements.
    \item \textbf{Adaptive strategy}: X-rays plus adaptivity achieve the smooth rate on smooth parts and detect singularities from chord-length profiles.
    \item Complete answer table over all body classes and measurement types.
  \end{itemize}
  \textbf{Convexity: required.}
